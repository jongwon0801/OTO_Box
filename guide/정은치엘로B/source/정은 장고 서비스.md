#### 정은치엘로 장고 서비스

```less
sudo nano /lib/systemd/system/django.service

[Unit]
Description=Django Service
After=network.target

[Service]
ExecStart=/home/pi/.virtualenvs/o2obox/bin/python /home/pi/Workspace/appleapp/manage.py runserver 0.0.0.0:8000
WorkingDirectory=/home/pi/Workspace/appleapp
StandardOutput=inherit
StandardError=inherit
Restart=always
User=pi

[Install]
WantedBy=multi-user.target
```

```less
# 서비스 파일 갱신
sudo systemctl daemon-reload  

# 서비스 자동실행 설정
sudo systemctl enable django.service

# 서비스 시작
sudo systemctl start django.service

# 서비스 상태
sudo systemctl status django.service

# 서비스 중지
sudo systemctl stop django.service

# 서비스 재시작
sudo systemctl restart django.service

# 서비스 상태확인
systemctl status django.service
```
